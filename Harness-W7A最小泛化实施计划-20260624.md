## 0. 计划口径与边界

### 0.1 核心定性

当前 Harness V1 的 `scripts/harness.js` 和 `workflows/weekly-canvas-task.json` 在多处硬编码 W6-A（taskId / subtask 列表 / reportRoot / 默认 fromSubtask / W6-A-01 imported-completed 特殊路径）。`weekly-canvas-task.json` 的 `taskId` 字段虽然是声明式，但 `loadWorkflowConfig` 在缓存路径上**单例化**了配置，导致 `pnpm harness step W7-A` 即使 taskId 解析正确，也会用到 W6-A 的 subtasks。

```js
function loadWorkflowConfig(taskId) {
  // HW7A-P1-002: cacheKey 含 taskId + HARNESS_ROOT，避免跨进程/跨 taskId 残留
  const cacheKey = `${HARNESS_ROOT}::${taskId}`;

  // 缓存命中：完全等价校验（含 _loaded 标志），复用 _workflowConfig 引用
  if (_workflowConfig &&
      _workflowConfigCacheKey === cacheKey &&
      _workflowConfig._loaded) {
    return _workflowConfig;
  }

  // 缓存失效：cacheKey miss → 完整重新解析 JSON → 替换 _workflowConfig 引用
  // 1. 路径解析：taskId-specific config 优先；找不到回退到 default
  const taskConfigPath = path.join(HARNESS_ROOT, 'workflows', `weekly-canvas-task-${taskId}.json`);
  const defaultConfigPath = path.join(HARNESS_ROOT, 'workflows', 'weekly-canvas-task.json');
  const configPath = fs.existsSync(taskConfigPath) ? taskConfigPath : defaultConfigPath;

  if (!fs.existsSync(configPath)) {
    throw new Error(`Workflow config not found for ${taskId}: ${configPath}`);
  }

  let raw;
  try {
    raw = fs.readFileSync(configPath, 'utf-8');
  } catch (err) {
    throw new Error(`Config read failed for ${taskId} at ${configPath}: ${err.message}`);
  }

  let parsed;
  try {
    parsed = JSON.parse(raw);
  } catch (err) {
    // P2-004 修复：JSON 解析失败抛明确错误，不静默 fallback
    throw new Error(`Config JSON parse failed for ${taskId} at ${configPath}: ${err.message}`);
  }

  // 关键：完全替换 _workflowConfig 模块级引用（HW7A-P1-002 闭环）
  _workflowConfig = parsed;
  _workflowConfigCacheKey = cacheKey;

  if (_workflowConfig.taskId !== taskId && taskId) {
    console.error(`Warning: Config taskId "${_workflowConfig.taskId}" differs from requested "${taskId}". Using config.`);
  }

  // HW7A-P1-003: taskTitle 必填（方案 A），config 漏写 throw
  if (!_workflowConfig.taskTitle) {
    throw new Error(
      `Config taskTitle is required for ${taskId}. ` +
      `Please add "taskTitle" field to ${configPath}.`
    );
  }

  _workflowConfig._loaded = true;
  return _workflowConfig;
}
```
